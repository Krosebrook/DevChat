export const getAllTools = (): ExtendedTool[] => {
  return toolCategories.flatMap(category => category.tools);
};

export const getToolsByCategory = (categoryId: string): ExtendedTool[] => {
  const category = toolCategories.find(cat => cat.id === categoryId);
  return category ? category.tools : [];
};

export const getToolById = (toolId: string): ExtendedTool | undefined => {
  return getAllTools().find(tool => tool.id === toolId);
};

export const getPopularTools = (): ExtendedTool[] => {
  return getAllTools().filter(tool => tool.popular);
};

export const getNewTools = (): ExtendedTool[] => {
  return getAllTools().filter(tool => tool.new);
};

export const getPremiumTools = (): ExtendedTool[] => {
  return getAllTools().filter(tool => tool.premium);
};
